#install
git clone https://github.com/emscripten-core/emsdk.git
cd emsdk/
./emsdk install latest
./emsdk activate latest

#prepare env (per term session)
source ./emsdk_env.sh --build=Release

#build
emcc -O3 test.c -s WASM=1 -o test.html

#serve curdir as http and open in browser (or --no browser)
emrun --port 8080 .
